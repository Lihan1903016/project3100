//Creating students Table in Database

CREATE TABLE students Of RUET (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  email VARCHAR(255) NOT NULL,
  password VARCHAR(255) NOT NULL,
  department VARCHAR(255) NOT NULL,
  series INT NOT NULL,
  roll INT NOT NULL
);

//Creating teachers Table in Database

CREATE TABLE Teachers_Of_RUET (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  email VARCHAR(255) NOT NULL,
  password VARCHAR(255) NOT NULL,
  department VARCHAR(255) NOT NULL
);


//Creating CT Marks Table in Database
